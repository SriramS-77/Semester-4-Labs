create or replace procedure instr(dept in varchar) is
cursor c1 is select * from instructor where dept_name=dept;
cursor c2 is select * from course where dept_name=dept;
begin
dbms_output.put_line('Instructors in ' || dept || ' department:');
for i in c1
loop
dbms_output.put_line(i.name);
end loop;
dbms_output.put_line('Courses offered by ' || dept || ' department:');
for i in c2
loop
dbms_output.put_line(i.title);
end loop;
end;
/


begin
instr('Finance');
end;
/


declare
dept varchar(20) := '&Department';
begin
instr(dept);
end;
/
